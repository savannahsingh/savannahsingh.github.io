            function updateToon(obj) {

                  push();

                  if ( int(random(10)) > 8) {
                   obj.nextX = int(random(obj.moveX.length));
                   obj.nextY = int(random(obj.moveY.length));
                  }

                  obj.posX += obj.moveX[obj.nextX];
                  obj.posY += obj.moveY[obj.nextY];


                  // console.log(obj.posX);

                   if (obj.posX > width) {
                     obj.posX = 0;
                   }

                   if (obj.posX < 0) {
                     obj.posX = width;
                   }

                   if (obj.posY > height) {
                     obj.posY = 0;
                   }

                   if (obj.posY < 0) {
                     obj.posY = height;
                   }

                   drawToon(obj);

                  pop();


            }
