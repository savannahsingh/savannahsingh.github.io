  void circle(float lox, float loy, int size, int x) {
    float a = size/x;
    for (int i = 0; i < x; i++) {
      fill(random(255),random(255),random(255));
      ellipse(lox, loy, size - i*a, size - i*a);
    }

  }

  void keyPressed() {

    if ( key == 'b' || key == 'B' ) {
      // example of toggle logic
      stroke(1);
      strokeWeight(3);
      line(mouseX, mouseY, pmouseX, pmouseY);
    } else {
      stroke(6);
      strokeWeight(10);
    }
  }
