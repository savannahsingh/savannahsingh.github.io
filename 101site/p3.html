            function isAMPM (hr) {
            let tod;
            if (hr < 12) {  //its AM
             tod = "AM";
             fill(230,255,90);
             ellipse(0,0, 50,50);


            } else { //otherwise PM
             tod = "PM";
             fill(255);
             ellipse(0,0,50,50);
             fill(0,0,10);
             ellipse(10,-10,30,30);
            }
            return tod;
            }
